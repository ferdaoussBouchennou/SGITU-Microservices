      \node[box,right=0.35cm of g10,fill=GreenPale,draw=GreenA,text=GreenA] (g4) {G4\\filtre rôles};
      \draw[arr] (u)--(g3)--(g10)--(g4);
    \end{tikzpicture}
  \end{center}

  \small
  \begin{itemize}
    \item \textbf{G3} : comptes + \texttt{ROLE\_G4\_OPERATOR}, \texttt{ROLE\_DISPATCHER}, \texttt{ROLE\_G4\_ADMIN}
    \item \textbf{G10} : login, émission JWT, routage \texttt{/api/g4/**}
    \item \textbf{G4} : re-vérifie signature (\texttt{JWT\_SECRET} partagé) + \texttt{SecurityConfig}
    \item Preuve : DISPATCHER $\rightarrow$ \texttt{POST /lignes} = \textbf{403}
  \end{itemize}
\end{frame}

\section{Conception}

\begin{frame}{Modèle métier — points clés UML}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \textbf{Séparation importante}
      \begin{itemize}
        \item \texttt{CoordinationEvent} : retard, déviation, panne
        \item \texttt{MissionIncidentImpact} : lien mission $\leftrightarrow$ ref. G9
        \item \texttt{VehiculeReferentiel} : UUID G7 (Kafka ou sync)
        \item \texttt{PendingNotification} : file si G5 down
      \end{itemize}
      \badgeorange{Règle 409 : 1 mission active / véhicule}
    \end{column}
    \begin{column}{0.50\textwidth}
      \figslide{figures/uml_classes_domaine_g4.png}{Classes domaine G4}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Flux à montrer à l'oral}
  \small
  \begin{enumerate}
    \item \textbf{Mission} : JWT DISPATCHER $\rightarrow$ POST mission (UUID véhicule) $\rightarrow$ 201 ou 409
    \item \textbf{G7} : topic \texttt{vehicule-positions} $\rightarrow$ log + déviation possible
    \item \textbf{G5 down} : POST notification $\rightarrow$ 202 \texttt{DEGRADED} + retry 30s
    \item \textbf{G9} : \texttt{incident-impacts} distinct de \texttt{/events}
  \end{enumerate}
  \figslide{figures/uml_seq_mission_create.png}{Séquence création mission}
\end{frame}

\begin{frame}{Chaos Monkey G5 — phrase clé}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.55\textwidth}
      \textbf{Démo soutenance}
      \begin{enumerate}
        \item Arrêter G5 (ou URL injoignable)
        \item \texttt{POST /api/notifications/send}
        \item \textbf{202 DEGRADED} — G4 reste UP
        \item \texttt{GET /api/g4/pending-notifications}
        \item Relancer G5 $\rightarrow$ retry
      \end{enumerate}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{alertblock}{À dire}
        « La commande métier ne tombe pas : la notification part en file et repart quand G5 revient. »
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}
